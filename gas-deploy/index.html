Queries

BigQuery SQL used to generate this report. DECLARE is inlined as string literals for BQ Agent compatibility.
Main analysis query — Revenue / Payer / Install share vs Impression share
Tables: prod_stream_view.cv (30d) · prod_stream_view.imp (7d) · Scan: ~2.48 TB · Rows: 1,000
WITH
revenue_30d AS (
  SELECT
    COALESCE(NULLIF(conv.req.device.model, ''), NULLIF(conv.req.device.hwv, ''), '(unknown)') AS device_label,
    conv.req.device.make AS device_make,
    SUM(conv.cv.revenue_usd.amount) AS revenue_usd,
    COUNT(*) AS revenue_events,
    COUNT(DISTINCT conv.req.device.ifa) AS paying_users
  FROM `focal-elf-631.prod_stream_view.cv` AS conv
  WHERE DATE(conv.timestamp) BETWEEN DATE_SUB(CURRENT_DATE(), INTERVAL 30 DAY) AND DATE_SUB(CURRENT_DATE(), INTERVAL 1 DAY)
    AND conv.api.product.id = 'wxAmPvDyZWpy4yeQ'
    AND conv.cv.revenue_usd.amount > 0
  GROUP BY 1, 2
),
imp_7d AS (
  SELECT
    COALESCE(NULLIF(req.device.model, ''), NULLIF(req.device.hwv, ''), '(unknown)') AS device_label,
    req.device.make AS device_make,
    COUNT(*) AS impressions,
    ROUND(SUM(imp.cost.billing.demand_charge_cost.usd.amount_micro) / 1e6, 2) AS spend_usd
  FROM `focal-elf-631.prod_stream_view.imp`
  WHERE DATE(timestamp) BETWEEN DATE_SUB(CURRENT_DATE(), INTERVAL 7 DAY) AND DATE_SUB(CURRENT_DATE(), INTERVAL 1 DAY)
    AND api.product.id = 'wxAmPvDyZWpy4yeQ'
  GROUP BY 1, 2
),
install_30d AS (
  -- cv used instead of pb to avoid 16TB scan
  SELECT
    COALESCE(NULLIF(conv.req.device.model, ''), NULLIF(conv.req.device.hwv, ''), '(unknown)') AS device_label,
    conv.req.device.make AS device_make,
    COUNT(*) AS installs
  FROM `focal-elf-631.prod_stream_view.cv` AS conv
  WHERE DATE(conv.timestamp) BETWEEN DATE_SUB(CURRENT_DATE(), INTERVAL 30 DAY) AND DATE_SUB(CURRENT_DATE(), INTERVAL 1 DAY)
    AND conv.api.product.id = 'wxAmPvDyZWpy4yeQ'
    AND LOWER(conv.cv.pb.event.name) LIKE '%install%'
  GROUP BY 1, 2
)
